            'a_recetacatdie' AS TableName        
           ,car_codigo        
       ,car_nombre        
           ,car_previo     
    FROM    cas_b_minuta c WITH ( NOLOCK )        
    INNER JOIN cas_b_minutadet b WITH ( NOLOCK ) ON c.min_cencos = b.mid_cecori        
                                                    AND c.min_codigo = b.mid_codigo        
    INNER JOIN dbo.b_receta AS cbr WITH ( NOLOCK ) ON b.mid_codrec = cbr.rec_codigo        
    INNER JOIN dbo.a_recetacatdie AS ar WITH ( NOLOCK ) ON cbr.rec_catdie = ar.car_codigo        
    INNER JOIN #tmp AS t WITH ( NOLOCK ) ON c.min_fecmin >= t.fechaini        
                                            AND c.min_fecmin <= t.fechafin        
                                            AND b.mid_cecori = t.Ceco

    INNER JOIN dbo.a_recetatippla AS ar WITH ( NOLOCK ) ON cbr.rec_tippla = ar.tip_codigo        
    INNER JOIN #tmp AS t WITH ( NOLOCK ) ON c.min_fecmin >= t.fechaini
