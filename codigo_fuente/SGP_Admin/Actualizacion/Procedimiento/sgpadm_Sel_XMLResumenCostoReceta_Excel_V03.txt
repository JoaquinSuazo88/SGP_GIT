alter PROC [dbo].[sgpadm_Sel_XMLResumenCostoReceta_Excel_V03]  
    (  
      @Ceco VARCHAR(20) ,  
      @CodRegimen INT ,  
      @CodServicio INT ,  
      @Fecha INT ,  
      @OpPrecio INT ,  
      @OpMuestraPre VARCHAR(1) ,  
      @XmlDietetica AS TEXT ,  
      @XmlPlato AS TEXT,  
      @AnoMesInicio as varchar(4),  
      @AnoMesFinal as varchar(4)   
    )  
  
AS   
  
    DECLARE @hdoc AS INTEGER                          
	declare @FechaSistema as int
	set @FechaSistema = CONVERT(INT, CONVERT(VARCHAR(8), GETDATE(), 112))

    EXEC sp_xml_preparedocument @hdoc OUTPUT, @XmlDietetica  
  
    SELECT  Die  
    INTO    #TmpDietetica  
    FROM    OPENXML (@hdoc, '/Dietetica/DetDietetica ', 1)                            
    WITH                            
    (                            
    Die INT   
    )                            
  
    EXEC sp_xml_preparedocument @hdoc OUTPUT, @XmlPlato  
  
    SELECT  Pla  
    INTO    #TmpPlato  
    FROM    OPENXML (@hdoc, '/Plato/DetPlato ', 1)                            
    WITH                            
  
    (                            
    Pla INT   
    )   
  
    CREATE INDEX ix_TmpDietetica_01 ON #TmpDietetica (Die)                                                                  
    CREATE INDEX ix_TmpPlato_01 ON #TmpPlato (Pla)                                                                  
  
    BEGIN                                                      
  
        SELECT  CONVERT(INT, 0) AS rec_codigo ,  
                CONVERT(VARCHAR(80), '') AS rec_nombre ,  
                CONVERT(INT, 0) AS rec_catdie ,  
                CONVERT(INT, 0) AS rec_tippla ,  
                CONVERT(INT, 0) AS rec_fecvig ,  
                CONVERT(VARCHAR(20), '') AS ing_codigo ,  
                CONVERT(FLOAT, 0) AS promedioreceta ,  
                CONVERT(int, 0) AS rec_indppr , 
                CONVERT(INT, 0) AS cod_uniReceta  
        INTO    #Paso_Final          
                                          
        CREATE INDEX ix_rec_codigo_01 ON #Paso_Final (rec_codigo)                                                                  
  
        DELETE  #Paso_Final                                                
  
        IF @OpMuestraPre = '1'   
  
            BEGIN                                                
  
                CREATE TABLE #sgpadm_Fun_CostoRecetaProducto  
                    (  
                      ing_codigo VARCHAR(20) COLLATE DATABASE_DEFAULT ,  
                      pro_codigo VARCHAR(20) COLLATE DATABASE_DEFAULT ,  
                      pro_nombre VARCHAR(100) COLLATE DATABASE_DEFAULT ,  
                      pro_facing FLOAT ,  
                      fi INT ,  
                      ft INT ,  
                      PMP FLOAT ,  
                      PMPIng FLOAT ,  
                      Convenio FLOAT ,  
                      ConvenioIng FLOAT ,  
                      Lista FLOAT ,  
                      ListaIng FLOAT  
                    )                                        
  
                INSERT  #sgpadm_Fun_CostoRecetaProducto  
                        ( ing_codigo ,  
                          pro_codigo ,  
                          pro_nombre ,  
                          pro_facing ,  
                          fi ,  
                          ft ,  
                          PMP ,  
                          PMPIng ,  
                          Convenio ,  
                          ConvenioIng ,  
                          Lista ,  
                          ListaIng                                
                        )  
  
                        EXEC dbo.PA_sgpadm_CostoRecetaProductoExcel_V03 @Ceco, @Fecha, @CodRegimen, @CodServicio, @OpPrecio, @AnoMesInicio, @AnoMesFinal   
                              
  
                SELECT  ing_codigo ,  
                        pro_codigo ,  
                        pro_nombre ,  
                        pro_facing ,  
                        fi ,  
                        ft ,  
                        PMP ,  
                        PMPIng ,  
                        Convenio ,  
                        ConvenioIng ,  
                        Lista ,  
                        ListaIng ,  
                        ( ISNULL(CASE WHEN @OpPrecio = 1 THEN Convenio  
                                      WHEN @OpPrecio = 2 THEN PMP  
                     ELSE Lista  
                        END, 0) / ISNULL(pro_facing, 0) ) AS pmp1  
                INTO    #sgpadm_Fun_CostoRecetaProducto2  
                FROM    #sgpadm_Fun_CostoRecetaProducto  
                WHERE   ( ISNULL(CASE WHEN @OpPrecio = 1 THEN Convenio  
                                      WHEN @OpPrecio = 2 THEN PMP  
                                      ELSE Lista  
                                 END, 0) ) > 0    
  
                CREATE INDEX ix_ing_codigo_01 ON #sgpadm_Fun_CostoRecetaProducto2 (ing_codigo)                            
  
                INSERT  INTO #Paso_Final  
                SELECT  br3.rec_codigo ,  
                        br3.rec_nombre ,  
                        br3.rec_catdie ,  
                        br3.rec_tippla ,  
                        br3.rec_fecvig ,  
                        bi.ing_codigo ,  
                        ROUND(ISNULL(AVG(fun.PMP1) * MAX(isnull(reemplazo.CantIngredienteCambio, br4.red_canpro) ), 0), 2) AS promedioreceta ,  
                        br3.rec_indppr ,  
						br3.cod_uniReceta  
                FROM    dbo.b_receta AS br3 WITH ( NOLOCK )  
                INNER JOIN dbo.b_recetadet AS br4 WITH ( NOLOCK ) ON br3.rec_codigo = br4.red_codigo  
				OUTER APPLY dbo.fn_ObtenerIngredienteReemplazoJerarquia(
				    @Ceco,
				    @CodRegimen,
				    br4.red_codigo,
				    br4.red_codpro,
				    br3.rec_tippla
				) AS reemplazo
                INNER JOIN dbo.b_ingrediente AS bi WITH ( NOLOCK ) ON ISNULL(reemplazo.IdIngredienteCambio, br4.red_codpro) = bi.ing_codigo  
                LEFT JOIN #sgpadm_Fun_CostoRecetaProducto2 AS fun ON fun.ing_codigo = bi.ing_codigo  
                                                                 AND fun.fi = @Fecha  
                WHERE br3.rec_catdie IN ( SELECT  Die FROM #TmpDietetica WITH ( NOLOCK ) )  
                AND   br3.rec_tippla IN ( SELECT  Pla FROM    #TmpPlato WITH ( NOLOCK ) )  
                AND br3.rec_indppr = 1 
                AND ( br3.rec_fecvig >  @fechasistema OR br3.rec_fecvig <= 0 )  
                GROUP BY br3.rec_codigo ,  
                         br3.rec_nombre ,  
                         br3.rec_catdie ,  
                         br3.rec_tippla ,  
                         br3.rec_fecvig ,  
                         br3.rec_indppr ,  
                         bi.ing_codigo ,  
                         br3.cod_uniReceta  
                ORDER BY br3.rec_nombre                            
  
                DROP TABLE #sgpadm_Fun_CostoRecetaProducto                            
  
                DROP TABLE #sgpadm_Fun_CostoRecetaProducto2                            
                               
            END                            
  
        ELSE   
  
            BEGIN                            
  
                INSERT  INTO #Paso_Final  
                SELECT  br3.rec_codigo ,  
                        br3.rec_nombre ,  
						br3.rec_catdie ,  
                        br3.rec_tippla ,  
                        br3.rec_fecvig ,  
                        bi.ing_codigo ,  
                        CONVERT(FLOAT, 0) AS promedioreceta ,  
						br3.rec_indppr ,  
                        br3.cod_uniReceta  
               FROM    dbo.b_receta AS br3 WITH ( NOLOCK )  
               INNER JOIN dbo.b_recetadet AS br4 WITH ( NOLOCK ) ON br3.rec_codigo = br4.red_codigo  
				OUTER APPLY dbo.fn_ObtenerIngredienteReemplazoJerarquia(
				    @Ceco,
				    @CodRegimen,
				    br4.red_codigo,
				    br4.red_codpro,
				    br3.rec_tippla
				) AS reemplazo
               INNER JOIN dbo.b_ingrediente AS bi WITH ( NOLOCK ) ON ISNULL(reemplazo.IdIngredienteCambio, br4.red_codpro) = bi.ing_codigo  
               WHERE   br3.rec_catdie IN ( SELECT  Die FROM #TmpDietetica WITH ( NOLOCK ) )  
               AND br3.rec_tippla IN ( SELECT  Pla FROM #TmpPlato WITH ( NOLOCK ) )  
               AND br3.rec_indppr = 1
               AND ( br3.rec_fecvig > CONVERT(INT, CONVERT(VARCHAR(8), GETDATE(), 112)) OR br3.rec_fecvig <= 0 )  
               GROUP BY br3.rec_codigo ,  
                        br3.rec_nombre ,  
                        br3.rec_catdie ,  
                        br3.rec_tippla ,  
                        br3.rec_fecvig ,  
                        br3.rec_indppr ,  
                        bi.ing_codigo ,  
                        br3.cod_uniReceta  
               ORDER BY br3.rec_nombre                                                                       
  
            END                            
  
     -- Selecciona las Ofertas- Tipo Negocio- Sellos- Zona- Estacionalidad que tiene el Casino  
  
        SELECT  pf.rec_codigo ,  
                pf.rec_nombre + ' ' + CONVERT(VARCHAR(20), pf.rec_codigo) AS rec_nombre ,  
                SUM(promedioreceta) AS promedioreceta ,  
                SUBSTRING(dbo.sgpadm_p_buscararbolcatdietetica(pf.rec_catdie), 1,  
                          LEN(dbo.sgpadm_p_buscararbolcatdietetica(pf.rec_catdie)) - 1) AS nom_catdie,  
                SUBSTRING(dbo.sgpadm_p_buscararboltipplato1(pf.rec_tippla), 1,  
                          LEN(dbo.sgpadm_p_buscararboltipplato1(pf.rec_tippla)) - 1) AS nom_tippla   
        FROM    #Paso_Final AS pf  
        INNER JOIN dbo.b_UnidadReceta AS bur WITH ( NOLOCK ) ON bur.Codigo_unidad = pf.cod_uniReceta  
        AND EXISTS ( SELECT 1 FROM dbo.b_receta_Oferta AS bro WITH ( NOLOCK )  
                     INNER JOIN dbo.b_Cliente_Oferta AS bco WITH ( NOLOCK ) ON bro.codigo_oferta = bco.codigo_oferta  
                     WHERE  bro.rec_codigo = pf.rec_codigo  
					 and    bro.Activo = '1'
					 and    bco.Activo = '1'
                     AND    bco.cli_codigo = @Ceco )  
         --and exists ( select 1 from b_recetatiponegocio as brtn with (nolock)   
         --             inner join a_tiponegocioreceta as atng with (nolock) on atng.IdTipoNegocio = brtn.IdTipoNegocio  
         --                                                                 and brtn.IdReceta = pf.rec_codigo  
         --                               and atng.Activo = '1'  
         --                                                                 and brtn.Activo = '1'  
         --             inner join b_ClientesTipoNegocio as bctn with (nolock) on bctn.IdTipoNegocio = atng.IDTipoNegocio  
         --                                                                   and bctn.IdCliente = @Ceco  
         --                                                                   and bctn.Activo = '1')  
         --and exists (select 1 from a_sellosreceta as asr with (nolock)   
         --            inner join b_receta as br with (nolock) on br.IdSellos = asr.IdSellos  
         --                                                   and br.rec_codigo = pf.rec_codigo  
         --            inner join b_ClientesSello as bcs with (nolock) on bcs.IdSello = asr.IdSellos   
         --                                                           and bcs.IdCliente = @Ceco
									--								and bcs.activo = '1' 
         --            where asr.Activo = '1')  
         --and exists (select 1 from b_recetazona as brz with (nolock)                                                
         --            inner join I_org_ceco as ioc with (nolock) on ioc.Id_orgcompra = brz.IdOrganizacionCompras  
         --                                                      and ioc.Id_ceco = @Ceco  
         --           and isnull(ioc.borrado,'') = ''   
    --            where brz.IdReceta = pf.rec_codigo  
         --            and   brz.activo = '1' )  
         --and exists (select 1 from a_estacionalidadreceta as aer with (nolock)   
         --            inner join b_recetaestacionalidad as bre with (nolock) on bre.IdEstacionalidad = aer.IdEstacionalidad  
         --                                                                  and bre.IdReceta = pf.rec_codigo  
         --                                                                  and bre.activo = '1'  
         --                                                                  and aer.activo = '1'  
         --                                                                  and convert(int,@AnoMesinicio) >= convert(int,aer.fecha_inicial)  
         --                                                                  and convert(int,@AnoMesFinal) <= convert(int,aer.fecha_final))  
        GROUP BY pf.rec_codigo ,  
                pf.rec_nombre ,  
                pf.rec_catdie ,  
                pf.rec_tippla  
        ORDER BY pf.rec_nombre                            
                                                
        DROP TABLE #Paso_Final                                                                
  
    END                            
  
    EXEC sp_xml_removedocument @hDoc                                   
  
    DROP TABLE #TmpDietetica                           
    DROP TABLE #TmpPlato
