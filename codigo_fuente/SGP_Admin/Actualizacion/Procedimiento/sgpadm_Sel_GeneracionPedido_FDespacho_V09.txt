alter PROC [dbo].[sgpadm_Sel_GeneracionPedido_FDespacho_V09]
    (
      @Ceco AS VARCHAR(10) ,
      @Fi AS VARCHAR(10) ,
      @Ft AS VARCHAR(10) ,
      @TipoPedido AS INT , --1:pedido PAP 2:proyectado 3: Pedido CD  
      @Familias AS XML = NULL ,
      @Incluir_Prod_CD AS INT ,
      @Incluir_Prod_PAP AS INT ,
      @IDPedido INT = NULL ,
      @OrgCompra VARCHAR(4) ,
      @fechalimite SMALLDATETIME ,
      @GeneraArrastreSaldo VARCHAR(1)
    )
AS
    SET NOCOUNT ON;
                                        
    DECLARE @ErrorMessage NVARCHAR(4000);
    DECLARE @Rows INT;
    DECLARE @IDPedido_NEW INT;
    DECLARE @IndI INT;
                              
    SET @IndI = 1;  
    SET @ErrorMessage = '';  
    SET @Rows = 0;  
                                  
    DECLARE @Ft_Aux AS INT;  
    DECLARE @FinDeSemana VARCHAR(1);  
    DECLARE @Ft_min INT;  
                                  
    DECLARE @Fi_D AS DATETIME;
    DECLARE @Fi_D_aux AS DATETIME;
    DECLARE @Ft_D AS DATETIME;
    DECLARE @FamiliasXmlData AS VARCHAR(MAX);
    SELECT  @FamiliasXmlData = CAST(@Familias AS NVARCHAR(MAX));
                                  
    SET @Fi_D = CONVERT(DATETIME, CONVERT(VARCHAR(10), @Fi));
    SET @Ft_D = CONVERT(DATETIME, CONVERT(VARCHAR(10), @Ft));
    SET @Fi_D_aux = @Fi_D;
                                  
    PRINT @Ft_D;
                                
    CREATE TABLE #tmp_fechas ( fecha INT );  
                        
    CREATE INDEX IX_FECHA_tmp_fechas ON #tmp_fechas (fecha);  
                 
    WHILE @Fi_D_aux <= @Ft_D
        BEGIN  
            INSERT  #tmp_fechas
                    ( fecha )
            VALUES  ( CONVERT(VARCHAR(10), @Fi_D_aux, 112) );  
            SET @Fi_D_aux = DATEADD(d, 1, @Fi_D_aux);  
        END;  
                                  
                                  
    SET @Ft_min = CONVERT(VARCHAR(8), DATEADD(d, 15,
                                              CONVERT(DATETIME, CONVERT(VARCHAR(8), @Ft))), 112);                                                                                            
                                               
    SET @FinDeSemana = '0';                                       
                                
    SELECT  @FinDeSemana = ISNULL(cli_blockmintrabajafinsemana, '0')
    FROM    b_clientes
    WHERE   cli_codigo = @Ceco;                                      
                                  
	/* Calculo Necesidades desde planificaciÃ³n */  
    SELECT  bm.min_fecmin ,
            bi.ing_codigo ,
            bm.min_codreg ,
            br2.red_codigo ,
            bp2.pro_codigo ,
            bm2.mid_numrac ,
            br2.red_canpro ,
            br.rec_basrac ,
            bp2.pro_facing,
			br.rec_tippla
    INTO    #Tmp_Minuta
    FROM    dbo.cas_b_minuta AS bm WITH ( NOLOCK )
            INNER JOIN dbo.cas_b_minutadet AS bm2 WITH ( NOLOCK ) ON bm.min_cecori = bm2.mid_cecori
                                                              AND bm.min_codigo = bm2.mid_codigo
            INNER JOIN dbo.b_clientes AS bc WITH ( NOLOCK ) ON bm.min_cecori = bc.cli_codigo
            INNER JOIN dbo.b_receta AS br WITH ( NOLOCK ) ON bm2.mid_codrec = br.rec_codigo
            INNER JOIN dbo.b_recetadet AS br2 WITH ( NOLOCK ) ON br.rec_codigo = br2.red_codigo
            INNER JOIN dbo.b_ingrediente AS bi WITH ( NOLOCK ) ON bi.ing_codigo = br2.red_codpro
            INNER JOIN dbo.b_productosing AS bp WITH ( NOLOCK ) ON bi.ing_codigo = bp.pri_coding
            INNER JOIN dbo.b_productos AS bp2 WITH ( NOLOCK ) ON bp2.pro_codigo = bp.pri_codpro
            INNER JOIN dbo.a_servicio AS as2 WITH ( NOLOCK ) ON bm.min_codser = as2.ser_codigo
    WHERE   bm.min_cecori = @Ceco
            AND bc.cli_activo = 1
            AND bc.cli_tipo = 0
            AND bm2.mid_tipmin = '1'
            AND bm2.mid_numrac > 0
            AND as2.ser_activo = '1'
            AND bm.min_fecmin >= @Fi
            AND bm.min_fecmin <= @Ft_min
            AND bp2.pro_codigo = ( SELECT TOP 1
                                            bp2.pri_codpro
                                   FROM     dbo.b_productosing AS bp2 WITH ( NOLOCK )
                                   WHERE    bp2.pri_coding = bi.ing_codigo
                                 )                               
    group by bm.min_fecmin ,
            bi.ing_codigo ,
            bm.min_codreg ,
            br2.red_codigo ,
            bp2.pro_codigo ,
            bm2.mid_numrac ,
            br2.red_canpro ,
            br.rec_basrac ,
            bp2.pro_facing,
			br.rec_tippla
			         
    CREATE INDEX ix_Tmp_Minuta_01 ON #Tmp_Minuta (red_codigo, ing_codigo, min_codreg);                                                
    CREATE INDEX ix_Tmp_Minuta_02 ON #Tmp_Minuta (ing_codigo);                                         
                          
	/* Actualizar tabla temporal minutas desde la tabla gramaje */  
    --UPDATE  #Tmp_Minuta
    --SET     #Tmp_Minuta.red_canpro = a.tgc_cantgr ,
    --        #Tmp_Minuta.ing_codigo = a.tgc_codins
    --FROM    b_tablagramajececo AS a WITH ( NOLOCK )
    --        INNER JOIN #Tmp_Minuta AS b WITH ( NOLOCK ) ON b.red_codigo = a.tgc_codrec
    --                         AND b.ing_codigo = a.tgc_coding
    --                                                       AND b.min_codreg = a.tgc_codreg
    --        INNER JOIN b_ingrediente AS c WITH ( NOLOCK ) ON a.tgc_codins = c.ing_codigo
    --        INNER JOIN dbo.b_clientes AS bc WITH ( NOLOCK ) ON bc.cli_codigo = a.tgc_ceco
    --WHERE   c.ing_activo = '1'
    --        AND bc.cli_tipo = 0
    --        AND bc.cli_activo = '1'
    --        AND bc.cli_codigo = @Ceco;  
                          
    UPDATE  #Tmp_Minuta
    SET     #Tmp_Minuta.red_canpro = isnull(reemplazo.CantIngredienteCambio, b.red_canpro) ,
            #Tmp_Minuta.ing_codigo = ISNULL(reemplazo.IdIngredienteCambio, b.ing_codigo) 
    FROM    #Tmp_Minuta AS b WITH ( NOLOCK )
			OUTER APPLY dbo.fn_ObtenerIngredienteReemplazoJerarquia(
				    @Ceco,
				    b.min_codreg,
				    b.red_codigo,
				    b.ing_codigo,
				    b.rec_tippla
		) AS reemplazo
            INNER JOIN b_ingrediente AS c WITH ( NOLOCK ) ON ISNULL(reemplazo.IdIngredienteOrigen, b.ing_codigo) = c.ing_codigo
    WHERE   c.ing_activo = '1'

	/* Borrar ingrediente que no estan considerado pedido */  
    DELETE  #Tmp_Minuta
    WHERE   ing_codigo IN (
            SELECT  ing_codigo
            FROM    b_ingrediente_pedido_Excepcion WITH ( NOLOCK ) );               
                             
	/* Calcular consumo ingrediente */        
    SELECT  tm.min_fecmin ,
            tm.ing_codigo ,
            bp2.pro_codigo ,
            ISNULL(SUM(( tm.mid_numrac * ( tm.red_canpro / tm.rec_basrac ) )),
                   0) AS Cant_ing ,
            CONVERT(DATETIME, CONVERT(VARCHAR(8), tm.min_fecmin), 112) AS Fecha_Consumo ,
            0 AS DiasHolgura ,
            CONVERT(DATETIME, NULL, 103) AS Fecha_en_Sitio
    INTO    #Tmp_Necesidad
    FROM    #Tmp_Minuta AS tm WITH ( NOLOCK )
            INNER JOIN dbo.b_ingrediente AS bi WITH ( NOLOCK ) ON bi.ing_codigo = tm.ing_codigo
            INNER JOIN dbo.b_productosing AS bp WITH ( NOLOCK ) ON bi.ing_codigo = bp.pri_coding
            INNER JOIN dbo.b_productos AS bp2 WITH ( NOLOCK ) ON bp2.pro_codigo = bp.pri_codpro
    WHERE   bp2.pro_codigo = ( SELECT TOP 1
                                        bp2.pri_codpro
                               FROM     dbo.b_productosing AS bp2 WITH ( NOLOCK )
                               WHERE    bp2.pri_coding = bi.ing_codigo
                             )
    GROUP BY tm.min_fecmin ,
            tm.ing_codigo ,
            bp2.pro_codigo;                                
                                  
	--	select * from #Tmp_Necesidad
	--return


    CREATE INDEX ix_Tmp_Necesidad_01 ON #Tmp_Necesidad (ing_codigo,min_fecmin);                                                
    CREATE INDEX ix_Tmp_Necesidad_02 ON #Tmp_Necesidad (ing_codigo);                                         
    CREATE INDEX ix_Tmp_Necesidad_03 ON #Tmp_Necesidad (pro_codigo);                                         
    CREATE INDEX ix_Tmp_Necesidad_04 ON #Tmp_Necesidad (DiasHolgura);                                         
    CREATE INDEX ix_Tmp_Necesidad_05 ON #Tmp_Necesidad (Fecha_en_Sitio);                                         
                                
	/* Actualiza dias de seguridad */                                      
    UPDATE  #Tmp_Necesidad
    SET     DiasHolgura = dbo.fn_sgpadm_Pro_TraerDiaSeguridad(bp.pro_codtip,
                                                              @Ceco)
    FROM    b_productos bp
    WHERE   #Tmp_Necesidad.pro_codigo = bp.pro_codigo;                                  
                                
	/* Actualiza Fecha_en_Sitio restando dias de seguridad a fecha de minuta*/                                      
    UPDATE  #Tmp_Necesidad
    SET     Fecha_en_Sitio = DATEADD(DD, ( DiasHolgura * -1 ), Fecha_Consumo)
    WHERE   DiasHolgura <> 0;                                  
                                        
    UPDATE  #Tmp_Necesidad
    SET     Fecha_en_Sitio = Fecha_Consumo
    WHERE   DiasHolgura = 0;                                  
                                        
	/* Actualiza Fecha_en_Sitio restando 1 dia a fecha de minuta si es domingo*/                                      
    UPDATE  #Tmp_Necesidad
    SET     Fecha_en_Sitio = DATEADD(DD, -1, Fecha_en_Sitio)
    WHERE   ( DATENAME(WEEKDAY, Fecha_en_Sitio) = 'Domingo'
              OR DATENAME(WEEKDAY, Fecha_en_Sitio) = 'Sunday'
            )
            AND @FinDeSemana = '0';                                  
            
	/* Actualiza Fecha_en_Sitio restando 1 dia MAS a fecha de minuta si es sabado*/                                      
    UPDATE  #Tmp_Necesidad
    SET     Fecha_en_Sitio = DATEADD(DD, -1, Fecha_en_Sitio)
    WHERE   ( DATENAME(WEEKDAY, Fecha_en_Sitio) = 'Sabado'
              OR DATENAME(WEEKDAY, Fecha_en_Sitio) = 'Saturday'
            )
            AND @FinDeSemana = '0';                                      
     
	/* Traer fecha proxima de la ruta */        
    SET @Ft_Aux = CONVERT(VARCHAR(8), DATEADD(mm, 2, @Ft_D), 112); --ISNULL(@Ft_Aux, @Ft)                            
                          
	/***********************************CONVENIO*******************************************************************/      
	/* CreaciÃ³n tabla temporal grupo despacho */      
    SELECT  br.Fecha_Despacho ,
            br.ID_Ceco ,
            brc.proveedor ,
            br.Idgrupodespacho ,
            CONVERT(VARCHAR(1), '0') AS Estado
    INTO    #rutagrupodespacho
    FROM    dbo.b_rutagrupodespacho AS br WITH ( NOLOCK )
            CROSS JOIN dbo.b_Rut_CD AS brc
    WHERE   br.ID_Ceco = @Ceco
            AND br.Fecha_Despacho >= @Fi
            AND br.Fecha_Despacho <= @Ft_Aux
            AND ISNULL(br.ID_proveedor, '') = '';      

	/* Crear tabla temporal ruta CD normal o bien grupo despacho*/	
    CREATE TABLE #b_rutadespachoCD
        (
          Fecha_Despacho INT ,
          ID_Ceco VARCHAR(10) COLLATE Database_Default ,
          ID_proveedor VARCHAR(10) COLLATE Database_Default ,
          ID_Ruta INT
        );

    CREATE INDEX ix_b_rutadespachoCD_01 ON #b_rutadespachoCD (Fecha_Despacho, ID_Ceco, ID_proveedor);

    IF EXISTS ( SELECT  1
                FROM    dbo.b_rutadespacho AS br WITH ( NOLOCK )
                WHERE   br.Fecha_Despacho >= @Fi
                        AND br.ID_Ceco = @Ceco
                        AND ISNULL(br.ID_proveedor, '') = '' )
        BEGIN

			INSERT  INTO #b_rutadespachoCD
                    ( Fecha_Despacho ,
                      ID_Ceco ,
                      ID_proveedor ,
                      ID_Ruta
			        )
                    SELECT  br.Fecha_Despacho ,
                            br.ID_Ceco ,

		convert(int,a.NumLinea) as NumLinea ,
		a.ing_codigo ,
		a.FamiliaProductoSAP ,
		a.total ,

		a.Proveedor

WHERE   a.activo = 1
		AND ( NOT EXISTS ( SELECT   1
							FROM     dbo.b_no_arrastre_saldo  AS BNAS with (nolock)
							WHERE    BNAS.fcs_CodMaterial = a.ProductoSAP )
				OR a.ing_codigo IN ( SELECT ing_codigo

										AND BAS.Saldo > 0 )

WHILE(@Counter IS NOT NULL

                                                        INNER JOIN dbo.b_PedidoCentralizado
