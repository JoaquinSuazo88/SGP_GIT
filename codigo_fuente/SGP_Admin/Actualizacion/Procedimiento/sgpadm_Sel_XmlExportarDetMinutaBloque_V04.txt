    (
      @MinBlo AS TEXT = NULL ,
      @OpPrecio AS INT ,
      @OrgCompra AS VARCHAR(4) ,
      @TipoInforme AS VARCHAR(1) ,
      @OpcionCosto AS VARCHAR(1),
	  @FechaIni    as int,
	  @FechaFin    as int
    )
AS 
    SET NOCOUNT ON                            
    --SET XACT_ABORT ON              
                  
    DECLARE @ErrorMessage NVARCHAR(4000)              
    DECLARE @hdoc AS INTEGER                                    
    EXEC sp_xml_preparedocument @hdoc OUTPUT, @MinBlo            
            
    BEGIN TRY                  
                                      
        SELECT  C AS Ceco ,
                B AS IndBloque
        INTO    #tmp_MinBlo
        FROM    OPENXML (@hdoc, '/MinBlo/C')                                    
      WITH (c VARCHAR(10),            
            b BIGINT)              

        DECLARE @DetReceta TABLE
				    (
					rec_codigo INT ,
					rec_nombre VARCHAR(80) COLLATE DATABASE_DEFAULT ,
					rec_tippla int ,
					red_codpro VARCHAR(20) COLLATE DATABASE_DEFAULT ,
					red_canpro INT ,
					red_numlin int,
					min_codreg INT ,
					cod_uniReceta int ,
					rec_LYD bit,
					PRIMARY KEY (rec_codigo, red_codpro, red_numlin, min_codreg)
					)                              
                  
        DECLARE @MyTableVar TABLE
            (
              ID_ORGCOMPRA VARCHAR(100) COLLATE DATABASE_DEFAULT ,
              min_cecori VARCHAR(10) COLLATE DATABASE_DEFAULT ,
              reg_codigo INT ,
              reg_nombre VARCHAR(100) COLLATE DATABASE_DEFAULT ,

              rec_codigo VARCHAR(100) COLLATE DATABASE_DEFAULT ,
              rec_nombre VARCHAR(100) COLLATE DATABASE_DEFAULT ,
              Ponderacion VARCHAR(100) COLLATE DATABASE_DEFAULT ,
              Raciones VARCHAR(100) COLLATE DATABASE_DEFAULT ,
              Comensales FLOAT ,
              mid_numlin float ,
              Cost_receta FLOAT
            )                              
                    
		SELECT 
			ROW_NUMBER() OVER (ORDER BY (SELECT NULL)) AS RowNum,
			ceco,
			IndBloque
		INTO #PasoTemp
		FROM dbo.#tmp_MinBlo

		DECLARE @i INT = 1
		DECLARE @Max INT = (SELECT COUNT(*) FROM #PasoTemp)

		declare @AuxCeco VARCHAR(10)
		DECLARE @ceco VARCHAR(10)
		DECLARE @idbloque INT

		set @AuxCeco=''

		WHILE @i <= @Max


            BEGIN                             

				print @i
				print @max

				SELECT 
					@ceco = ceco,
					@idbloque = IndBloque
				FROM #PasoTemp
				WHERE RowNum = @i

				if @ceco<> @AuxCeco
					
					begin              
						-- Ahora puedes usar @DetReceta como quieras
						DELETE FROM @DetReceta

						insert into @DetReceta
						SELECT  br.rec_codigo,
								br.rec_nombre,
								br.rec_tippla,
								br2.red_codpro,
								br2.red_canpro,
								br2.red_nroite,
								cbm.min_codreg,
								br.cod_uniReceta ,
								ISNULL(br.rec_LYD, 0) AS rec_LYD
						FROM    dbo.cas_b_MinutaBloque AS bmb WITH ( NOLOCK )
                        INNER JOIN dbo.cas_b_minuta AS cbm WITH ( NOLOCK ) ON bmb.Ceco = cbm.min_cecori
                                                                              AND bmb.ID_Bloque = cbm.ID_Bloque
                                                                              AND cbm.min_fecmin >= CONVERT(INT, CONVERT(VARCHAR(8), bmb.FechaDesde, 112))
                                                                              AND cbm.min_fecmin <= CONVERT(INT, CONVERT(VARCHAR(8), bmb.FechaHasta, 112))
                        INNER JOIN dbo.cas_b_minutadet AS cbm2 WITH ( NOLOCK ) ON bmb.Ceco = cbm2.mid_cecori
                                                                                  AND cbm.min_codigo = cbm2.mid_codigo
                        INNER JOIN dbo.b_receta AS br WITH ( NOLOCK ) ON cbm2.mid_codrec = br.rec_codigo
                        INNER JOIN dbo.b_recetadet AS br2 WITH ( NOLOCK ) ON br.rec_codigo = br2.red_codigo
                        INNER JOIN dbo.b_clientes AS bc WITH ( NOLOCK ) ON bmb.Ceco = bc.cli_codigo  
						WHERE   bmb.Ceco = @ceco                           
						and     cbm.min_fecmin>= @FechaIni
						and     cbm.min_fecmin<= @FechaFin
						group by br.rec_codigo,
								br.rec_nombre,
								br.rec_tippla,
								br2.red_codpro,
								br2.red_canpro,
								br2.red_nroite,
								cbm.min_codreg,
								br.cod_uniReceta ,
								br.rec_LYD

						update @DetReceta
						set red_codpro = ISNULL(reemplazo.IdIngredienteCambio, a.red_codpro),
							red_canpro = isnull(reemplazo.CantIngredienteCambio, a.red_canpro)
						from @DetReceta as a
						OUTER APPLY dbo.fn_ObtenerIngredienteReemplazoJerarquia(
									@ceco,
									a.min_codreg,
									a.rec_codigo,
									a.red_codpro,
									a.rec_tippla
								) AS reemplazo

						-- 1️⃣ Eliminar la tabla si ya existe
						IF OBJECT_ID('tempdb..#sgpadm_Fun_CostoMinutaProducto') IS NOT NULL
							DROP TABLE #sgpadm_Fun_CostoMinutaProducto;

						IF OBJECT_ID('tempdb..#sgpadm_Fun_CostoMinutaProducto2') IS NOT NULL
							DROP TABLE #sgpadm_Fun_CostoMinutaProducto2;

							
		                CREATE TABLE #sgpadm_Fun_CostoMinutaProducto
				            (
						      ing_codigo VARCHAR(20) COLLATE DATABASE_DEFAULT ,
		                      pro_codigo VARCHAR(20) COLLATE DATABASE_DEFAULT ,
				              pro_nombre VARCHAR(100) COLLATE DATABASE_DEFAULT ,
						      pro_facing FLOAT ,
							fi INT ,
							ft INT ,
							PMP FLOAT ,
							Convenio FLOAT ,
							Lista FLOAT
							)                              
                                            
						INSERT  #sgpadm_Fun_CostoMinutaProducto
								( ing_codigo ,
								pro_codigo ,
								pro_nombre ,
								pro_facing ,
								fi ,
								ft ,
								PMP ,
								Convenio ,
								Lista                              
								)
								EXEC dbo.PA_sgpadm_CostoMinutaProductoCeco_V01 @fechaini, @Fechafin, @Ceco, @OpPrecio                                  
                                      
		                SELECT  ing_codigo ,
				                pro_codigo ,
						        pro_nombre ,
								pro_facing ,
								fi ,
								ft ,
								PMP ,
								Convenio ,
								Lista ,
								( ISNULL(CASE WHEN @OpPrecio = 1 THEN Convenio
										      WHEN @OpPrecio = 2 THEN PMP
											ELSE Lista
										END, 0) / ISNULL(pro_facing, 0) ) AS pmp1
						INTO    #sgpadm_Fun_CostoMinutaProducto2
						FROM    #sgpadm_Fun_CostoMinutaProducto
						WHERE   ( ISNULL(CASE WHEN @OpPrecio = 1 THEN Convenio
								              WHEN @OpPrecio = 2 THEN PMP
										      ELSE Lista
										END, 0) ) > 0                                                 

					   set @AuxCeco = @ceco

				end;
				 
                SELECT  cbm.min_fecmin ,
                        cbm.min_codreg ,
                        cbm.min_indblo ,
                        cbm2.mid_numlin ,
                        ISNULL(cbm2.mid_numrac, 0) AS mid_numrac ,
                        ISNULL(cbm2.mid_porrac, 0) AS mid_porrac ,
                        ISNULL(cbm2.mid_modmina, '') AS mid_modmina ,
                        ISNULL(cbm2.mid_modminb, '') AS mid_modminb ,
                        ISNULL(mid_tiprec, 0) AS mid_tiprec ,
                        ae.ess_codigo ,
						case when isnull(cbm2.mid_desest,'') <> '' then cbm2.mid_desest
                             else ISNULL(ae.ess_nombre, '') end AS ess_nombre ,
                        ISNULL(ae.ess_agrupacionestructura, 0) AS ess_agrupacionestructura ,
                        ISNULL(mingrpest.mge_ponderaciontotal, -1) AS mge_ponderaciontotal ,
                        br.rec_codigo ,
                        ISNULL(br.rec_nombre, '') AS rec_nombre ,
                        br.red_codpro AS ing_codigo ,
                        br.red_canpro AS red_canbru ,
                        br.cod_uniReceta ,

                INTO    #PasoMinutaBloque
                FROM    dbo.cas_b_MinutaBloque AS bmb WITH ( NOLOCK )
                        INNER JOIN dbo.cas_b_minuta AS cbm WITH ( NOLOCK ) ON bmb.Ceco = cbm.min_cecori
                                                                              AND bmb.ID_Bloque = cbm.ID_Bloque
                                                                              AND cbm.min_fecmin >= CONVERT(INT, CONVERT(VARCHAR(8), bmb.FechaDesde, 112))
                                                                              AND cbm.min_fecmin <= CONVERT(INT, CONVERT(VARCHAR(8), bmb.FechaHasta, 112))

                        INNER JOIN dbo.a_estservicio AS ae WITH ( NOLOCK ) ON ae.ess_codigo = cbm2.mid_estser
						                                                  and ae.ess_codser = cbm.min_codser
                        LEFT  JOIN ( SELECT  DISTINCT
                                            bm.mge_cencos ,
                                            bm.mge_grupoestructura ,
                                            bm.mge_id_bloque ,
                                            bm.mge_ponderaciontotal
                                     FROM   dbo.cas_b_minutagrupoestructura AS bm WITH ( NOLOCK )
                                   ) mingrpest ON mingrpest.mge_cencos = bmb.Ceco
                                                  AND mingrpest.mge_id_bloque = bmb.ID_Bloque
                                                  AND mingrpest.mge_grupoestructura = ae.ess_agrupacionestructura
                        INNER JOIN @DetReceta AS br ON cbm2.mid_codrec = br.rec_codigo and cbm.min_codreg = br.min_codreg
                        INNER JOIN dbo.b_clientes AS bc WITH ( NOLOCK ) ON bmb.Ceco = bc.cli_codigo
                WHERE   bmb.ID_Bloque = @idbloque
                        AND bmb.Ceco = @ceco                           
                              
                                            
                SELECT  tg.min_fecmin ,

                        tg.mid_numlin ,
                        tg.rec_codigo ,
                        tg.mid_numrac ,
                        tg.mid_porrac ,
                        tg.mid_modmina ,
                        tg.mid_modminb ,
                        tg.mid_tiprec ,
                        tg.ess_codigo ,
                        tg.ess_nombre ,
                        tg.ess_agrupacionestructura ,
                        tg.mge_ponderaciontotal ,
                        tg.rec_nombre ,
                        tg.ing_codigo AS codigoingrediente ,
                        tg.min_codreg ,
                        ISNULL(AVG(fun.PMP1) * MAX(tg.red_canbru), 0) AS promedioreceta ,
                        tg.cod_uniReceta ,
                        tg.rec_LYD
                INTO    #tmp_final
                FROM    #PasoMinutaBloque AS tg
                        INNER JOIN dbo.b_ingrediente AS bi WITH ( NOLOCK ) ON tg.ing_codigo = bi.ing_codigo
                   LEFT JOIN #sgpadm_Fun_CostoMinutaProducto2 AS fun ON fun.ing_codigo = tg.ing_codigo

                GROUP BY tg.min_fecmin ,
                        tg.min_indblo ,
                        tg.mid_numlin ,
                        tg.rec_codigo ,
                        tg.mid_numrac ,
                        tg.mid_porrac ,
                        tg.mid_modmina ,
                        tg.mid_modminb ,
                        tg.mid_tiprec ,
                        tg.ess_codigo ,
                        tg.ess_nombre ,
                        tg.ess_agrupacionestructura ,
                        tg.mge_ponderaciontotal ,
                        tg.rec_nombre ,
                        tg.ing_codigo ,
                        tg.min_codreg ,
                        tg.cod_uniReceta ,
                        tg.rec_LYD
                ORDER BY tg.mid_numlin ,
                        tg.min_fecmin ,
                        tg.mge_ponderaciontotal DESC                                             

                INSERT  @MyTableVar
                        ( ID_ORGCOMPRA ,
                          min_cecori ,
                          reg_codigo ,
                          reg_nombre ,
                          ser_codigo ,
                          ser_nombre ,
                          ess_codigo ,
                          ess_nombre ,
                          min_fecmin ,
                          rec_codigo ,
                          rec_nombre ,
                          Ponderacion ,
                          Raciones ,
                          Comensales ,
                          mid_numlin ,
                          Cost_receta                        
                        )
                        SELECT  '' ,
                                @Ceco ,
                                ar.reg_codigo ,
                                ar.reg_nombre ,
                                as2.ser_codigo ,
                                as2.ser_nombre ,
                                t.ess_codigo ,

                                CASE WHEN bur.DescripcionCorta IS NULL THEN LTRIM(RTRIM(ISNULL(t.rec_nombre, '')))
                                     ELSE LTRIM(RTRIM(ISNULL(t.rec_nombre, ''))) + ' ['
                                          + LTRIM(RTRIM(ISNULL(bur.DescripcionCorta, ''))) + ']'
                                END AS rec_nombre ,
                                ISNULL(t.mid_porrac,0) AS mid_porrac,
                                ROUND(ISNULL(t.mid_numrac,0),0) AS mid_numrac ,
                                0 ,
                                t.mid_numlin ,
                                CONVERT(NUMERIC(18, 2), SUM(ISNULL(t.promedioreceta, 0))) AS promedioreceta             
                                --'ISNULL(dbo.fn_sgpadm_Pro_TraerAporteCalorias(t.rec_codigo, @Cur_Cencos, t.min_codreg), 0) AS AporteNut            
                        FROM    #tmp_final t
                                LEFT JOIN dbo.b_UnidadReceta AS bur WITH ( NOLOCK ) ON t.cod_uniReceta = bur.Codigo_unidad
                                INNER JOIN dbo.CAS_b_MinutaBloque AS cbmb WITH ( NOLOCK ) ON cbmb.ID_Bloque = @IdBloque
                                                                                             AND cbmb.Ceco = @Ceco
                                INNER JOIN dbo.a_regimen AS ar WITH ( NOLOCK ) ON ar.reg_codigo = cbmb.Regimen
                                INNER JOIN dbo.a_servicio AS as2 WITH ( NOLOCK ) ON as2.ser_codigo = cbmb.Servicio

                                                        @Aux_Reg_Nombre ,

                                                        0 ,
                                                        '' ,
                                                        0 ,
                                                        0 ,
                                                        @Cur_Min_RacTeo ,
                                                        99999 ,
                                                        CASE WHEN ISNULL(@Aux_Min_RacTeo, 0) = 0 THEN 0
                                                             ELSE ISNULL(@Aux_Tot_Dia / ISNULL(@Aux_Min_RacTeo, 0), 0)
