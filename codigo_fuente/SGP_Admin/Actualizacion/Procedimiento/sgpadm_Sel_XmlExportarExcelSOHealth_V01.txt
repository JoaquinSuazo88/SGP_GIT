alter PROC [dbo].[sgpadm_Sel_XmlExportarExcelSOHealth_V01]
    (
      @XmlServicio AS TEXT = NULL ,
      @Ceco AS VARCHAR(10) ,
      @Regimen AS INT ,
      @FecIni AS INT ,
      @FecFin AS INT
    )
AS 

SET NOCOUNT ON                        
SET XACT_ABORT ON              
                
DECLARE @hdoc AS INTEGER                                      
                                      
EXEC sp_xml_preparedocument @hdoc OUTPUT, @XmlServicio                                      
                                      
SELECT  Ser
INTO    #TmpServicio
FROM    OPENXML (@hdoc, '/Servicio/Ser ', 1)                                      
WITH                                      
  (                                      
  Ser  INT            
  )                                      
                
CREATE INDEX #TmpServicio_01 ON #TmpServicio (Ser)            

SELECT  br.rec_codigo
      , br.rec_nombre
	  , br.rec_nomfan
	  , br.rec_catdie
	  , convert(varchar(1000),'') as [Categoria Dietetica]
	  , br.rec_tippla
	  ,	convert(varchar(1000),'') as [Tipo de Plato]
INTO    #tempreceta
FROM    dbo.cas_b_minuta AS cbm WITH ( NOLOCK )
        INNER JOIN dbo.cas_b_minutadet AS cbm2 WITH ( NOLOCK ) ON cbm.min_cecori = cbm2.mid_cecori
                                                                   AND cbm.min_codigo = cbm2.mid_codigo
        INNER JOIN dbo.b_receta AS br WITH ( NOLOCK ) ON cbm2.mid_codrec = br.rec_codigo
        INNER JOIN dbo.b_recetadet AS br2 WITH ( NOLOCK ) ON br.rec_codigo = br2.red_codigo
        INNER JOIN dbo.b_productosing AS bp WITH ( NOLOCK ) ON br2.red_codpro = bp.pri_coding
        INNER JOIN dbo.b_productos AS bp2 WITH ( NOLOCK ) ON bp.pri_codpro = bp2.pro_codigo
        INNER JOIN dbo.b_receta_Oferta AS bro WITH ( NOLOCK ) ON bro.rec_codigo = br.rec_codigo
        INNER JOIN dbo.b_Cliente_Oferta AS bco WITH ( NOLOCK ) ON bco.codigo_oferta = bro.codigo_oferta
        INNER JOIN dbo.b_clientes AS bc WITH ( NOLOCK ) ON bc.cli_codigo = bco.cli_codigo
                                                           AND bc.cli_codigo = cbm.min_cecori
        INNER JOIN dbo.b_Ofertas AS bo WITH ( NOLOCK ) ON bo.Codigo_oferta = bro.codigo_oferta
		inner join #TmpServicio as tm on tm.ser = cbm.min_codser 
		WHERE br.rec_indppr = 1
        AND bco.cli_codigo = @Ceco
		and cbm.min_codreg = @Regimen
        AND cbm.min_fecmin >= @Fecini
        AND cbm.min_fecmin <= @FecFin
        AND cbm2.mid_tipmin = 1
GROUP BY br.rec_codigo
       , br.rec_nombre
	   , br.rec_nomfan
	   , br.rec_catdie
	   , br.rec_tippla

/* Actualizar categoria dietetica */
 update #tempreceta
 set [Categoria Dietetica] = ISNULL(SUBSTRING(dbo.sgpadm_p_buscararbolcatdietetica(a.rec_catdie), 1, LEN(dbo.sgpadm_p_buscararbolcatdietetica(a.rec_catdie)) - 1), '')
 from #tempreceta as a
 inner join a_recetacatdie as b with (nolock) on b.car_codigo = a.rec_catdie

/* Actualizar tipo de plato */
 update #tempreceta
 set [Tipo de Plato] = ISNULL(SUBSTRING(dbo.sgpadm_p_buscararboltipplato1(a.rec_tippla), 1, LEN(dbo.sgpadm_p_buscararboltipplato1(a.rec_tippla) )- 1), '')
 from #tempreceta as a
 inner join a_recetatippla as b with (nolock) on b.tip_codigo = a.rec_tippla

        br.red_pctcoc AS red_pctcoc ,

		convert(float,0) as Calorias,
		convert(float,0) as Proteinas,

		convert(float,0) as Fibras,

		convert(float,0) as [Vitamina B12],
		convert(float,0) as [Folato],

		convert(float,0) as [Vitamina C],
		convert(float,0) as [Vitamina E],
		convert(float,0) as [Calcio],
		convert(float,0) as [Cobre],
		convert(float,0) as [Hierro],
		convert(float,0) as [Magnesio],
		convert(float,0) as [Fosforo],
		convert(float,0) as [Potasio],
		convert(float,0) as [Selenio],
		convert(float,0) as [Sodio],
		convert(float,0) as [Zinc],

        INNER JOIN dbo.b_recetadet AS br WITH ( NOLOCK ) ON br.red_codigo = tr.rec_codigo
        INNER JOIN dbo.b_ingrediente AS bi WITH ( NOLOCK ) ON bi.ing_codigo = br.red_codpro
                                                              AND bi.ing_activo = 1
                                                              AND bi.ing_indppr = 1

        INNER JOIN dbo.b_productos AS bp2 WITH ( NOLOCK ) ON bp2.pro_codigo = bp.pri_codpro
                                                             AND bp2.pro_activo = 1
                                                             AND bp2.pro_indppr = 1

                                    ELSE dr.red_pctnut
                               END,
        red_pctapr = CASE WHEN reemplazo.IdIngredienteOrigen is not null and reemplazo.IdIngredienteOrigen <> reemplazo.IdIngredienteCambio THEN bi2.ing_pctapr
                                    ELSE dr.red_pctapr
                               END,
		red_pctcoc = CASE WHEN reemplazo.IdIngredienteOrigen is not null and reemplazo.IdIngredienteOrigen <> reemplazo.IdIngredienteCambio THEN bi2.ing_pctcoc
        ELSE dr.red_pctcoc

SELECT cbm.min_cecori as [Ceco],
       bc.cli_nombre as [Nombre Ceco],
	   substring(convert(varchar(8),cbm.min_fecmin),7,2) + '/' + substring(convert(varchar(8),cbm.min_fecmin),5,2) + '/' + substring(convert(varchar(8),cbm.min_fecmin),1,4)  as [Fecha],
	   cbm.min_codreg as [Regimen],
	   ar.reg_nombre as [Nombre Regimen],
	   cbm.min_codser as [Servicio],
	   ase.ser_nombre as [Nombre Servicio],
	   cbm2.mid_codrec as [Código Receta],
       dr.[Nombre receta] as [Nombre Receta],
	   dr.[Nombre fantasia] as [Nombre fantasia],
	   dr.[Categoria Dietetica] as [Categoria Dietetica],
	   dr.[Tipo de Plato] as [Tipo de Plato],
        dr.[Ing. Origen] ,
        bi.ing_nombre AS [Nombre Ing. Origen] ,
        dr.[Cant Origen] as [Gramos Bruto],
		 ( ISNULL(( dr.red_pctnut / 100 ), 0) * ( dr.[Cant Origen] ) ) AS [Gramos Neto],
        ( ( ( dr.red_pctapr / 100 ) * dr.[Cant Origen] ) * ( dr.red_pctcoc / 100 ) ) AS [Gramo Neto Servido] ,
        ( dr.[Cant Origen] * dr.red_pctnut ) / 100 AS [Gramo Neto Nutricional] ,
        au.unm_nomcor as [Unidad Medida],
		Humedad,
		Calorias,
		Proteinas,
		Hidratos,
		Fibras,
		Lipidos,
		[Grasa Total],
		[Ac. Grasos Sat.],
		[Ac. Grasos Mon],
		[Ac. Grasos Poli],
		[Colesterol],
		[N6],
		[N3],
		[Caroteno],
		[Retinol],
		[Vit. A Tot Re],
		[Vitamina B1],
		[Vitamina B2],
		[Niacina ( B3)],
		[Vitamina B6],
		[Vitamina B12],
		[Folato],
		[Ac. Pantot (B5)],
		[Vitamina C],
		[Vitamina E],
		[Calcio],
		[Cobre],
		[Hierro],
		[Magnesio],
		[Fosforo],
		[Potasio],
		[Selenio],
		[Sodio],
		[Zinc],
		[A.c Graso Trans.],
		[Azucares Totales]
FROM    dbo.cas_b_minuta AS cbm WITH ( NOLOCK )
INNER JOIN dbo.cas_b_minutadet AS cbm2 WITH ( NOLOCK ) ON cbm.min_cecori = cbm2.mid_cecori
                                                      AND cbm.min_codigo = cbm2.mid_codigo
inner join #TmpServicio as tm on tm.ser = cbm.min_codser 
inner join b_clientes as bc with (nolock) on bc.cli_codigo = cbm.min_cecori
inner join a_regimen as ar with (nolock) on ar.reg_codigo = cbm.min_codreg
inner join a_servicio as ase with (nolock) on ase.ser_codigo = cbm.min_codser
inner join dbo.#DetalleReceta AS dr WITH ( NOLOCK ) on dr.[Receta] = cbm2.mid_codrec
INNER JOIN dbo.b_ingrediente AS bi WITH ( NOLOCK ) ON dr.[Ing. Origen] = bi.ing_codigo
INNER JOIN dbo.a_unidadmed AS au WITH ( NOLOCK ) ON au.unm_codigo = bi.ing_unimed
WHERE bc.cli_codigo = @Ceco
and   cbm.min_codreg = @Regimen
AND   cbm.min_fecmin >= @Fecini
AND   cbm.min_fecmin <= @FecFin
AND   cbm2.mid_tipmin = 1
order by regimen,
         servicio,
		 Fecha,
		 cbm2.mid_numlin
