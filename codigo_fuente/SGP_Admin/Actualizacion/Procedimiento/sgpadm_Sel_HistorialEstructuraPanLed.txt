CREATE PROCEDURE [dbo].[sgpadm_Sel_HistorialEstructuraPanLed] ( @Ceco VARCHAR(10) )
AS 
    SET NOCOUNT ON          
    SET XACT_ABORT ON  
    

    SELECT DISTINCT
            bc.cli_codigo AS Ceco ,
            bc.cli_nombre AS CecoDesc ,
            ar.reg_codigo AS Regimen ,
            ar.reg_nombre AS RegimenDesc ,
            as2.ser_codigo AS Servicio ,
            as2.ser_nombre AS ServicioDesc
    FROM    dbo.homologacionPantalla_Led AS hd WITH ( NOLOCK )
            INNER JOIN dbo.b_clientes AS bc WITH ( NOLOCK ) ON bc.cli_codigo = hd.IdCeco
            INNER JOIN dbo.a_regimen AS ar WITH ( NOLOCK ) ON ar.reg_codigo = hd.IdRegimen
            INNER JOIN dbo.a_servicio AS as2 WITH ( NOLOCK ) ON as2.ser_codigo = hd.IdServicio
    WHERE   hd.IdCeco = @Ceco
    ORDER BY ar.reg_codigo ,
            as2.ser_codigo
