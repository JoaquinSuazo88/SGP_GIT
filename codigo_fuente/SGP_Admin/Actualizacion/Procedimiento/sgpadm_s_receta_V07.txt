CREATE PROC [dbo].[sgpadm_s_receta_V07]
    (
      @aux INT ,
      @codigo INT ,
      @descripcion VARCHAR(50) ,
      @catdie INT ,
      @tippla INT ,
      @anomes INT ,
      @usuario VARCHAR(20)
    )

AS 

    DECLARE @fechoy INT, @nreg INT                                            

    SELECT  @fechoy = 0 ,
            @nreg = 0                      

    DECLARE @GlosaSql NVARCHAR(MAX)                      

    DECLARE @CalCosto AS VARCHAR(1)                

    SET @CalCosto = 'N'                    

    SELECT  @CalCosto = ISNULL(par_valor, 'N')
    FROM    a_param
    WHERE   par_codigo = 'cosproingr'                

    SELECT  @fechoy = CONVERT(INT, CONVERT(VARCHAR(4), DATEPART(yy, GETDATE())) + SUBSTRING('00' + CONVERT(VARCHAR(2), DATEPART(MM,GETDATE())),
	                          DATALENGTH(CONVERT(VARCHAR(2), DATEPART(MM, GETDATE()))) + 1, 2) + SUBSTRING('00'                                                                  + CONVERT(VARCHAR(2), DATEPART(dd,
                              GETDATE())), DATALENGTH(CONVERT(VARCHAR(2), DATEPART(dd, GETDATE()))) + 1, 2))                     
                
    IF @aux IN ( 19, 2, 7, 11, 12, 13, 14, 15, 16, 17, 21, 24, 26 ) 

        BEGIN                

			--> Rutina buscar más una familia categoria dietetica                
            DECLARE @eof VARCHAR(1)                

            SET @eof = '1'                

            SELECT  0 AS catdie
            INTO    #TmpCatDie
            FROM    a_recetacatdie WITH ( NOLOCK )                
            DELETE  #TmpCatDie                 

            IF @catdie > 0 

                BEGIN                

    INSERT  INTO #TmpCatDie
                    SELECT  @catdie AS catdie                 

                    BEGIN                

                        WHILE @eof = '1' 

                            IF EXISTS ( SELECT DISTINCT car_codigo
                                        FROM    a_recetacatdie WITH ( NOLOCK )
                                        WHERE   car_previo = @catdie ) 

                                BEGIN                

                                    INSERT  INTO #TmpCatDie
                                    SELECT DISTINCT car_codigo
                                    FROM    a_recetacatdie WITH ( NOLOCK )
                                    WHERE   car_previo = @catdie                

                                    SELECT  @catdie = car_codigo
                                    FROM    a_recetacatdie WITH ( NOLOCK )
                                    WHERE   car_previo = @catdie                

                                END                

                            ELSE

                                BEGIN   

                                    SET @eof = '0'

                                END

                    END       

                END            

            ELSE

                BEGIN                

                    INSERT  INTO #TmpCatDie
                    SELECT  car_codigo
                    FROM    a_recetacatdie WITH ( NOLOCK )                

                END                
                 
			--> Rutina buscar más una familia tipo plato                

            SELECT  0 AS tippla
            INTO    #TmpTipPla
            FROM    a_recetatippla WITH ( NOLOCK )                

            DELETE  #TmpTipPla                 

            IF @tippla > 0

                BEGIN                

                    SET @eof = '1'                 

                    INSERT  INTO #TmpTipPla
                    SELECT  @tippla AS tippla                

                    BEGIN                

                        WHILE @eof = '1' 

                            IF EXISTS ( SELECT DISTINCT tip_codigo
                                        FROM    a_recetatippla WITH ( NOLOCK )
                                        WHERE   tip_previo = @tippla ) 

                                BEGIN                

                                    INSERT  INTO #TmpTipPla
                                    SELECT DISTINCT tip_codigo
                                    FROM    a_recetatippla WITH ( NOLOCK )
                                    WHERE   tip_previo = @tippla                

                                    SELECT  @tippla = tip_codigo
                                    FROM    a_recetatippla WITH ( NOLOCK )
                                    WHERE   tip_previo = @tippla      

                                END

                            ELSE 

                                BEGIN                

                                    SET @eof = '0'                

                                END                

                    END

                END                

            ELSE 

             BEGIN                

                    INSERT  INTO #TmpTipPla
                    SELECT  tip_codigo AS tippla
                    FROM    a_recetatippla WITH ( NOLOCK )                

                END

        END
                
    IF @aux = 2 

        SELECT  COUNT(*) AS nreg
        FROM    b_receta WITH ( NOLOCK )
        WHERE   rec_catdie IN ( SELECT  catdie FROM #TmpCatDie WITH ( NOLOCK ) )
        AND rec_tippla IN ( SELECT  tippla FROM #TmpTipPla WITH ( NOLOCK ) )
        AND UPPER(rec_nombre) LIKE @descripcion
        AND rec_tiprec = '0'
                                            
    IF @aux = 3 

        SELECT  rec_codigo ,
                rec_nombre ,
rec_fecvig ,
                rec_metpre ,
                rec_gruvul ,
                rec_indppr ,
                rec_activo
        FROM    b_receta WITH ( NOLOCK )
  WHERE   ( rec_catdie = @catdie OR @catdie = 0 )
        AND     ( rec_tippla = @tippla OR @tippla = 0 )
        AND     UPPER(rec_nombre) LIKE @descripcion
        AND     rec_tiprec = '0'
  ORDER BY rec_nombre                                            
                                            
    IF @aux = 4 

        SELECT  rec_metpre
        FROM    b_receta WITH ( NOLOCK )
        WHERE   rec_codigo = @codigo                                      
                                            
    IF @aux = 5 

        SELECT  rec_codigo
        FROM    b_receta WITH ( NOLOCK )
        ORDER BY rec_codigo DESC                                            
                                            
    IF @aux = 6 

        SELECT  red_codpro
        FROM    b_recetadet WITH ( NOLOCK )
        WHERE   red_codigo = @codigo                                            
                                            
    IF @aux = 7 

        SELECT  rec_codigo ,
                rec_nombre ,
                rec_fecvig ,
                rec_metpre ,
                rec_gruvul ,
                rec_indppr ,
                rec_activo
        FROM    b_receta WITH ( NOLOCK )
        WHERE   rec_catdie IN ( SELECT  catdie FROM #TmpCatDie WITH ( NOLOCK ) )
        AND     rec_tippla IN ( SELECT  tippla FROM #TmpTipPla WITH ( NOLOCK ) )
        AND     rec_tiprec = '0'
        ORDER BY rec_nombre                                            
                                            
    IF @aux = 8 

        SELECT  a.rec_codigo ,
                a.rec_nombre ,
                a.rec_nomfan ,
                a.rec_catdie ,
                a.rec_tippla ,
                a.rec_basrac ,
                a.rec_fecvig ,
                b.car_nombre ,
                c.tip_nombre ,
                a.rec_indppr ,
                a.rec_canser ,
                a.rec_activo ,
                a.cod_uniReceta ,
                ISNULL(SUBSTRING(dbo.sgpadm_p_buscararbolcatdietetica(a.rec_catdie), 1,
                                 LEN(dbo.sgpadm_p_buscararbolcatdietetica(a.rec_catdie)) - 1), '') AS CatDietetica ,
                ISNULL(SUBSTRING(dbo.sgpadm_p_buscararboltipplato1(a.rec_tippla), 1,
                LEN(dbo.sgpadm_p_buscararboltipplato1(a.rec_tippla) )- 1), '') AS TipPla,
				isnull(acr.IdCosto, -1) as IdCosto,
				isnull(atipr.Idtipoingprincipal, -1) as Idtipoingprincipal,
				isnull(atipr.Nombretipoingprincipal,'') as Nombretipoingprincipal,
				isnull(amcr.IdMetodoCoccion, -1) as IdMetodoCoccion,
				isnull(amcr.NombreMetodoCoccion,'') as NombreMetodoCoccion,
				isnull(accr.IdCategorizacionCompleja, -1) as IdCategorizacionCompleja,
				isnull(aicgr.IdIngCruceGarnitura, -1) as IdIngCruceGarnitura,
				isnull(aicgr.NombreIngCruceGarnitura,'') as NombreIngCruceGarnitura,
				isnull(aemr.IdEfectoMeteorizante, -1) as IdEfectoMeteorizante,
				isnull(asr.IdSellos, -1) as IdSellos,
				isnull(atcr.IdTiempoCoccion, -1) as IdTiempoCoccion,
				convert(varchar(5),ISNULL(atcr.Hora, ''),8) as tiempococcionHora,
				isnull(atht.IdTiempoHh, -1) as IdTiempoHh,
				convert(varchar(5),ISNULL(atht.Hora, ''),8) as tiempohhHora,
				isnull(ac.IdColor, -1) as IdColor,
				isnull(ac.NombreColor,'') as NombreColor,
				isnull(aesr.IdEtiquetadoSello, -1) as IdEtiquetadoSello,
				isnull(atipr2.Idtipoingprincipal, -1) as IdSegundoingprincipal,
				isnull(atipr2.Nombretipoingprincipal,'')  as NombreSegtipoingprincipal,
				isnull(aec.IdEquipamientoCoccion, -1) as IdEquipamientoCoccion,
				isnull(aps.Idparametrosalsa, -1) as Idparametrosalsa,
				isnull(aia.IdIntegraAMD,1) as IdIntegraAMD 
        FROM   b_receta a WITH ( NOLOCK ) 
        inner join a_recetacatdie b WITH ( NOLOCK ) on a.rec_catdie = b.car_codigo
        inner join a_recetatippla c WITH ( NOLOCK ) on a.rec_tippla = c.tip_codigo
		left join a_costoreceta acr WITH ( NOLOCK ) on acr.IdCosto = a.IdCosto
		                                           and isnull(acr.Activo, '') = '1'
		left join a_tipoingredienteprincipalreceta atipr WITH ( NOLOCK ) on atipr.Idtipoingprincipal = a.Idtipoingprincipal
		                                                                and isnull(atipr.Activo, '') = '1'
		left join a_metodococcionreceta amcr WITH ( NOLOCK ) on amcr.IdMetodoCoccion = a.IdMetodoCoccion
		                                                    and isnull(amcr.Activo, '') = '1'
		left join a_categorizacioncomplejareceta accr WITH ( NOLOCK ) on accr.IdCategorizacionCompleja = a.IdCategorizacionCompleja
		                                                              and isnull(accr.Activo, '') = '1'
		left join a_ingredientecrucegarniturareceta aicgr WITH ( NOLOCK ) on aicgr.IdIngCruceGarnitura = a.IdIngCruceGarnitura
		                                                                 and isnull(aicgr.Activo, '') = '1'
        left join a_efectometeorizantereceta aemr WITH ( NOLOCK ) on aemr.IdEfectoMeteorizante = a.IdEfectoMeteorizante
		                                                         and isnull(aemr.Activo, '') = '1'
		left join a_sellosreceta asr WITH ( NOLOCK ) on asr.IdSellos = a.IdSellos
		                                            and isnull(asr.Activo, '') = '1'
		left join a_etiquetadoselloreceta aesr WITH ( NOLOCK ) on aesr.IdEtiquetadoSello = a.IdEtiquetadoSello
		                                            and isnull(aesr.Activo, '') = '1'
		left join a_tiempococcionreceta atcr WITH ( NOLOCK ) on atcr.IdTiempoCoccion = a.IdTiempoCoccion
		                                                    and isnull(atcr.Activo, '') = '1'
		left join a_tiempohhreceta atht WITH ( NOLOCK ) on atht.IdTiempoHh = a.IdTiempoHh
		                                               and isnull(atht.Activo, '') = '1'
		left join a_color ac WITH ( NOLOCK ) on ac.IdColor = a.IdColor
		                                    and isnull(ac.Activo, '') = '1'
		left join a_tipoingredienteprincipalreceta atipr2 WITH ( NOLOCK ) on atipr2.Idtipoingprincipal = a.IdSegundoIngredientePrincipal
		                                                                and isnull(atipr2.Activo, '') = '1'
		left join a_equipamientococcion aec WITH ( NOLOCK ) on aec.IdEquipamientoCoccion = a.IdEquipamientoCoccion
		                                                                and isnull(aec.Activo, '') = '1'

		left join a_parametrosalsa aps WITH ( NOLOCK ) on aps.IdParametroSalsa = a.IdParametroSalsa
		                                                                and isnull(aps.Activo, '') = '1'
		left join a_integraAMD aia with (nolock) on aia.IdIntegraAMD = a.IdIntegraAMD
												and aia.activo = '1'
        WHERE a.rec_codigo = @codigo                                     

    IF @aux = 9 

        SELECT  b.red_codigo ,
                b.red_nroite ,
                b.red_codpro ,
                b.red_canpro ,
                b.red_cospro ,
                b.red_pctapr ,
                b.red_pctcoc ,
                b.red_pctnut ,
                ( ( b.red_pctnut / 100 ) * ( b.red_canpro ) ) AS canneta ,
                ( ( b.red_pctnut / 100 ) * ( b.red_canpro ) ) AS cangvneta ,
                ( b.red_canpro * c.ing_precos ) AS precos ,
                ( ( ( b.red_pctapr / 100 ) * b.red_canpro ) * ( b.red_pctcoc / 100 ) ) AS canservida ,
                c.ing_nombre ,
                c.ing_indgrv ,
                c.ing_precos ,
                d.unm_codigo ,
                d.unm_nomcor ,
                a.rec_indppr
        FROM    b_receta as a WITH ( NOLOCK )
        inner join b_recetadet as b WITH ( NOLOCK ) on   b.red_codigo = a.rec_codigo
        inner join b_ingrediente as c WITH ( NOLOCK ) on b.red_codpro = c.ing_codigo
                                                     AND   c.ing_activo = '1'                
        inner join a_unidadmed as d WITH ( NOLOCK ) on c.ing_unimed = d.unm_codigo
		WHERE b.red_codigo = @codigo
                                                                                        
    IF @aux = 10 

        SELECT  rec_gruvul
        FROM    b_receta WITH ( NOLOCK )
        WHERE   rec_codigo = @codigo                                            
                                            
    IF @aux = 11 

        SELECT a.rec_codigo ,
               a.rec_nombre ,
               a.rec_indppr ,
               SUM(b.red_canpro * c.ing_precos) AS cosrec
        FROM   b_receta a WITH ( NOLOCK )
        inner join b_recetadet b WITH ( NOLOCK ) on  a.rec_codigo = b.red_codigo
        inner join b_ingrediente c WITH ( NOLOCK ) on b.red_codpro = c.ing_codigo
                                                  AND c.ing_activo = '1'
        WHERE ( a.rec_fecvig > @fechoy OR a.rec_fecvig <= 0 )
        AND a.rec_catdie IN ( SELECT catdie FROM #TmpCatDie WITH ( NOLOCK ) )
        AND a.rec_tippla IN ( SELECT tippla FROM #TmpTipPla WITH ( NOLOCK ) )
        GROUP BY a.rec_codigo ,
                a.rec_nombre ,
                a.rec_indppr
        ORDER BY a.rec_nombre                                              
                                  
    IF @aux = 12 

        SELECT DISTINCT
                a.rec_codigo ,
                a.rec_nombre ,
                b.rec_codrec ,
                a.rec_indppr
        FROM    b_receta a WITH ( NOLOCK )
        LEFT JOIN b_recetacasino b WITH ( NOLOCK ) ON a.rec_codigo = b.rec_codrec
        WHERE   b.rec_codrec IS NULL
        AND a.rec_catdie IN ( SELECT catdie FROM #TmpCatDie WITH ( NOLOCK ) )
        AND a.rec_tippla IN ( SELECT tippla FROM #TmpTipPla WITH ( NOLOCK ) )
        AND a.rec_tiprec = '0'
        ORDER BY a.rec_nombre                                            
                                            
    IF @aux = 13 

        SELECT DISTINCT
                a.rec_codigo ,
                a.rec_nombre ,
                b.rec_codrec ,
                a.rec_indppr
        FROM    b_receta a WITH ( NOLOCK )
        LEFT JOIN b_recetacasino b WITH ( NOLOCK ) ON a.rec_codigo = b.rec_codrec
        WHERE   a.rec_catdie IN ( SELECT catdie FROM #TmpCatDie WITH ( NOLOCK ) )
        AND     a.rec_tippla IN ( SELECT tippla FROM #TmpTipPla WITH ( NOLOCK ) )
        AND     a.rec_tiprec = '0'
        ORDER BY a.rec_nombre                
                                            
    IF @aux = 14 

        BEGIN                                            
                                            
            SELECT  @nreg = COUNT(DISTINCT a.rec_codigo)
            FROM    b_receta a WITH ( NOLOCK )
            LEFT JOIN b_recetacasino b WITH ( NOLOCK ) ON a.rec_codigo = b.rec_codrec
            WHERE   b.rec_codrec IS NULL
            AND     a.rec_catdie IN ( SELECT catdie FROM #TmpCatDie WITH ( NOLOCK ) )
            AND     a.rec_tippla IN ( SELECT tippla FROM #TmpTipPla WITH ( NOLOCK ) )
            AND     a.rec_tiprec = '0'
            AND     UPPER(a.rec_nombre) LIKE @descripcion                                            
                                            
            SELECT DISTINCT
                    a.rec_codigo ,
                    a.rec_nombre ,
                    b.rec_codrec ,
                    @nreg AS nreg ,
                    a.rec_indppr
            FROM    b_receta a WITH ( NOLOCK )
            LEFT JOIN b_recetacasino b WITH ( NOLOCK ) ON a.rec_codigo = b.rec_codrec
            WHERE   b.rec_codrec IS NULL
            AND     a.rec_catdie IN ( SELECT catdie FROM #TmpCatDie WITH ( NOLOCK ) )
            AND     a.rec_tippla IN ( SELECT tippla FROM #TmpTipPla WITH ( NOLOCK ) )
            AND     a.rec_tiprec = '0'
            AND     UPPER(a.rec_nombre) LIKE @descripcion
            ORDER BY a.rec_nombre            
      
        END                                            
                                            
    IF @aux = 15 

        BEGIN                                            
            
            SELECT  @nreg = COUNT(DISTINCT a.rec_codigo)
            FROM    b_receta a WITH ( NOLOCK )
            LEFT JOIN b_recetacasino b WITH ( NOLOCK ) ON a.rec_codigo = b.rec_codrec
            WHERE   a.rec_catdie IN ( SELECT catdie FROM #TmpCatDie WITH ( NOLOCK ) )
            AND     a.rec_tippla IN ( SELECT tippla FROM #TmpTipPla WITH ( NOLOCK ) )
            AND     a.rec_tiprec = '0'
            AND     UPPER(a.rec_nombre) LIKE @descripcion                                            
                                                                                        
            SELECT DISTINCT
                    a.rec_codigo ,
                    a.rec_nombre ,
                    b.rec_codrec ,
                    @nreg AS nreg ,
                    a.rec_indppr
            FROM    b_receta a WITH ( NOLOCK )
            LEFT JOIN b_recetacasino b WITH ( NOLOCK ) ON a.rec_codigo = b.rec_codrec
            WHERE   a.rec_catdie IN ( SELECT catdie FROM #TmpCatDie WITH ( NOLOCK ) )
            AND     a.rec_tippla IN ( SELECT tippla FROM #TmpTipPla WITH ( NOLOCK ) )
            AND     a.rec_tiprec = '0'
            AND     UPPER(a.rec_nombre) LIKE @descripcion
            ORDER BY a.rec_nombre                                            
                                            
        END                                            
                                            
    IF @aux = 16 

        BEGIN                                            
                                            
            SELECT  @nreg = COUNT(DISTINCT a.rec_codigo)
            FROM    b_receta a WITH ( NOLOCK )
            LEFT JOIN b_recetacasino b WITH ( NOLOCK ) ON a.rec_codigo = b.rec_codrec
            WHERE   b.rec_codrec IS NULL
            AND     a.rec_catdie IN ( SELECT catdie FROM #TmpCatDie WITH ( NOLOCK ) )
            AND     a.rec_tippla IN ( SELECT tippla FROM #TmpTipPla WITH ( NOLOCK ) )
            AND     a.rec_tiprec = '0'
            AND UPPER(a.rec_codigo) LIKE @descripcion                                            
                                            
            SELECT DISTINCT
                    a.rec_codigo ,
                    a.rec_nombre ,
                    b.rec_codrec ,
                    a.rec_indppr
            FROM    b_receta a WITH ( NOLOCK )
            LEFT JOIN b_recetacasino b WITH ( NOLOCK ) ON a.rec_codigo = b.rec_codrec
            WHERE   b.rec_codrec IS NULL
            AND     a.rec_catdie IN ( SELECT catdie FROM #TmpCatDie WITH ( NOLOCK ) )
            AND     a.rec_tippla IN ( SELECT tippla FROM #TmpTipPla WITH ( NOLOCK ) )
            AND     a.rec_tiprec = '0'
            AND     UPPER(a.rec_codigo) LIKE @descripcion
            ORDER BY a.rec_codigo                                            
                                            
        END                            
                                            
    IF @aux = 17 

        BEGIN                                            
                                            
            SELECT  @nreg = COUNT(DISTINCT a.rec_codigo)
            FROM    b_receta a WITH ( NOLOCK )
            LEFT JOIN b_recetacasino b WITH ( NOLOCK ) ON a.rec_codigo = b.rec_codrec
            WHERE   a.rec_catdie IN ( SELECT catdie FROM #TmpCatDie WITH ( NOLOCK ) )
            AND     a.rec_tippla IN ( SELECT tippla FROM #TmpTipPla WITH ( NOLOCK ) )
            AND     a.rec_tiprec = '0'
            AND     UPPER(a.rec_codigo) LIKE @descripcion                                            
 
            SELECT DISTINCT
         a.rec_codigo ,
                    a.rec_nombre ,
     b.rec_codrec ,
       a.rec_indppr
            FROM    b_receta a WITH ( NOLOCK )
            LEFT JOIN b_recetacasino b WITH ( NOLOCK ) ON a.rec_codigo = b.rec_codrec
           WHERE   a.rec_catdie IN ( SELECT catdie FROM #TmpCatDie WITH ( NOLOCK ) )
            AND     a.rec_tippla IN ( SELECT tippla FROM #TmpTipPla WITH ( NOLOCK ) )
            AND     a.rec_tiprec = '0'
            AND     UPPER(a.rec_codigo) LIKE @descripcion
            ORDER BY a.rec_codigo                                 

        END                                            
                                            
IF @aux = 18      -- sgpadm_s_receta 18, 2599, '', 2, 0, 201108, 'adm'                                      

        BEGIN                                            
                                            
            SELECT  b.red_codigo ,
                    isnull(b.red_nroite,0) as  red_nroite,
                    isnull(b.red_codpro,'') as red_codpro,
                    isnull(b.red_canpro,0) as red_canpro,
                    isnull(b.red_cospro,0) as red_cospro,
                    ISNULL(b.red_pctapr, 0) AS red_pctapr ,
                    ISNULL(b.red_pctcoc, 0) AS red_pctcoc ,
                    ISNULL(b.red_pctnut, 0) AS red_pctnut ,
                    ( ISNULL(( b.red_pctnut / 100 ), 0) * ( b.red_canpro ) ) AS canneta ,
                    ( ISNULL(( b.red_pctnut / 100 ), 0) * ( b.red_canpro ) ) AS cangvneta ,
                    CONVERT(FLOAT, 0) AS precos ,
                    ( ( ( b.red_pctapr / 100 ) * b.red_canpro ) * ( b.red_pctcoc / 100 ) ) AS canservida ,
                    isnull(c.ing_nombre,'') as ing_nombre,
                    isnull(c.ing_indgrv,0) as ing_indgrv,
                    isnull(c.ing_precos,0) as ing_precos,
                    isnull(c.ing_indppr,0) as ing_indppr,
                    isnull(d.unm_codigo,0) as unm_codigo,
                    isnull(d.unm_nomcor,'') as unm_nomcor,
                    isnull(a.rec_indppr,0) as rec_indppr,
                    ISNULL(red_IndentificadorIngSumaTablaGramaje, '0') AS red_IndentificadorIngSumaTablaGramaje,
					isnull(c.Huella_Carbono,0) as HuellaCarbono
            INTO    #paso
            FROM    b_receta a WITH ( NOLOCK )
            inner join b_recetadet b WITH ( NOLOCK ) on b.red_codigo = a.rec_codigo
            inner join b_ingrediente c WITH ( NOLOCK ) on b.red_codpro = c.ing_codigo
                                                      AND c.ing_activo = '1'                
            inner join a_unidadmed d WITH ( NOLOCK ) on c.ing_unimed = d.unm_codigo
            WHERE  b.red_codigo = @codigo
                
            IF @CalCosto = 'N' 

                BEGIN                
                
                    UPDATE  #paso
