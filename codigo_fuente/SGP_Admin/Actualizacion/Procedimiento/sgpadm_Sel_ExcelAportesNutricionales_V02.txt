create  PROC [dbo].[sgpadm_Sel_ExcelAportesNutricionales_V02]
    (
      @XmlIngrediente AS TEXT   
    )
AS 
    SET NOCOUNT ON          
    SET XACT_ABORT ON        
  
    DECLARE @hdoc AS INTEGER                                  
                                  
    EXEC sp_xml_preparedocument @hdoc OUTPUT, @XmlIngrediente  
                                  
    SELECT  Ingr
    INTO    #TmpIngr
    FROM    OPENXML (@hdoc, '/Ingredientes/Ing ', 1)                                  
    WITH                                  
    (                                  
    Ingr VARCHAR(20)      
    )                               
            
    SELECT  DISTINCT
            ing_codigo ,
            ing_nombre ,
            ing_pctapr AS [% Aprovechamiento] ,
            ing_pctcoc AS [% Cocción] ,
            ing_pctnut AS [% Nutricional] ,
            ing_facnut AS [Factor Nutricional] ,
			isnull(Huella_Carbono,0) as [Huella Carbono],

            CONVERT(FLOAT, 0) AS [Humedad] ,
            CONVERT(INT, 2) AS [Calorias1] ,

            CONVERT(INT, 6) AS [Lipidos1] ,
            CONVERT(FLOAT, 0) AS [Lipidos] ,

            CONVERT(FLOAT, 0) AS [Ac. Grasos Poli] ,

            CONVERT(FLOAT, 0) AS [Colesterol] ,
            CONVERT(INT, 12) AS [N61] ,
            CONVERT(FLOAT, 0) AS [N6] ,
            CONVERT(INT, 13) AS [N31] ,

    WHERE   bi.ing_activo = '1'

            --AND bp2.pro_ctacon = '410001'

    WHERE   #Tmp_IngAporte.[Humedad1] = bp.pnu_codapo

    WHERE   #Tmp_IngAporte.[Calorias1] = bp.pnu_codapo
            AND #Tmp_IngAporte.ing_codigo = bp.pnu_codpro   
    
    UPDATE  #Tmp_IngAporte
    SET     [Proteinas] = bp.pnu_canapo

    WHERE   #Tmp_IngAporte.[Ac. Grasos Poli1] = bp.pnu_codapo

    UPDATE  #Tmp_IngAporte

    WHERE   #Tmp_IngAporte.[N31] = bp.pnu_codapo
            AND #Tmp_IngAporte.ing_codigo = bp.pnu_codpro     
                                    
    UPDATE  #Tmp_IngAporte
    SET     [Caroteno] = bp.pnu_canapo

    WHERE   #Tmp_IngAporte.[Caroteno1] = bp.pnu_codapo
            AND #Tmp_IngAporte.ing_codigo = bp.pnu_codpro     
    
    UPDATE  #Tmp_IngAporte
    SET     [Retinol] = bp.pnu_canapo
    FROM    dbo.b_productonut AS bp
    WHERE   #Tmp_IngAporte.[Retinol1] = bp.pnu_codapo
            AND #Tmp_IngAporte.ing_codigo = bp.pnu_codpro    
    
    UPDATE  #Tmp_IngAporte
    SET     [Vit. A Tot Re] = bp.pnu_canapo
    FROM    dbo.b_productonut AS bp
    WHERE   #Tmp_IngAporte.[Vit. A Tot Re1] = bp.pnu_codapo
            AND #Tmp_IngAporte.ing_codigo = bp.pnu_codpro    
    
    UPDATE  #Tmp_IngAporte
    SET     [Vitamina B1] = bp.pnu_canapo
    FROM    dbo.b_productonut AS bp
    WHERE   #Tmp_IngAporte.[Vitamina B11] = bp.pnu_codapo
            AND #Tmp_IngAporte.ing_codigo = bp.pnu_codpro    
    
    UPDATE  #Tmp_IngAporte
    SET     [Vitamina B2] = bp.pnu_canapo
    FROM    dbo.b_productonut AS bp
    WHERE   #Tmp_IngAporte.[Vitamina B21] = bp.pnu_codapo
            AND #Tmp_IngAporte.ing_codigo = bp.pnu_codpro    
    
    UPDATE  #Tmp_IngAporte
    SET     [Niacina ( B3)] = bp.pnu_canapo

    UPDATE  #Tmp_IngAporte

    WHERE   #Tmp_IngAporte.[Ac. Pantot (B5)1] = bp.pnu_codapo
            AND #Tmp_IngAporte.ing_codigo = bp.pnu_codpro    
    
    UPDATE  #Tmp_IngAporte
    SET     [Vitamina C] = bp.pnu_canapo
    FROM    dbo.b_productonut AS bp
    WHERE   #Tmp_IngAporte.[Vitamina C1] = bp.pnu_codapo
            AND #Tmp_IngAporte.ing_codigo = bp.pnu_codpro    
    
    UPDATE  #Tmp_IngAporte
    SET     [Vitamina E] = bp.pnu_canapo
    FROM    dbo.b_productonut AS bp
    WHERE   #Tmp_IngAporte.[Vitamina E1] = bp.pnu_codapo
            AND #Tmp_IngAporte.ing_codigo = bp.pnu_codpro    
    
    UPDATE  #Tmp_IngAporte
    SET     [Calcio] = bp.pnu_canapo
    FROM    dbo.b_productonut AS bp
    WHERE   #Tmp_IngAporte.[Calcio1] = bp.pnu_codapo
            AND #Tmp_IngAporte.ing_codigo = bp.pnu_codpro    
            
    UPDATE  #Tmp_IngAporte
    SET     [Cobre] = bp.pnu_canapo
    FROM    dbo.b_productonut AS bp
    WHERE   #Tmp_IngAporte.[Cobre1] = bp.pnu_codapo
            AND #Tmp_IngAporte.ing_codigo = bp.pnu_codpro    
    
    UPDATE  #Tmp_IngAporte
    SET     [Hierro] = bp.pnu_canapo
    FROM    dbo.b_productonut AS bp
    WHERE   #Tmp_IngAporte.[Hierro1] = bp.pnu_codapo
            AND #Tmp_IngAporte.ing_codigo = bp.pnu_codpro            
    
    UPDATE  #Tmp_IngAporte
    SET     [Magnesio] = bp.pnu_canapo

            [% Cocción] ,
            [% Nutricional] ,
            [Factor Nutricional] ,
			[Huella Carbono],

            [Humedad] ,
            [Calorias1] AS [Cód. Calorias] ,

            [Hidratos] ,
            [Fibras1] AS [Cód. Fibras] ,
            [Fibras] ,
            [Lipidos1] AS [Cód. Lipidos] ,
            [Lipidos] ,
            [Grasa Total1] AS [Cód. Grasa Total] ,
            [Grasa Total] ,
            [Ac. Grasos Sat.1] AS [Cód. Ac. Grasos Sat.] ,
            [Ac. Grasos Sat.] ,
